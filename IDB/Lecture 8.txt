SELECT JOB, SAL,
              DECODE(JOB, 'ANALYST',  SAL*1.1,
                          'CLERK',   SAL*1.15,
                          'MANAGER', SAL*1.20, 
                          'PRESIDENT', SAL) "REVISED_SALARY"
FROM   EMP

select ename from emp

select ename, sal, comm, 
            decode(ename, 'MARTIN', sal+comm,
                           'MILLER', sal*2,
                           'BLAKE', sal+(nvl(comm,0)*2)) "New Salary"
from emp

from emp
